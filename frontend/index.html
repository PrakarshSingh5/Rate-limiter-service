Rate Limiting Algorithms

Choose the right algorithm for your use case

Token Bucket

Allows bursts of traffic while maintaining an average rate. Tokens are added at a fixed rate, and each request consumes a token.

✓ Handles traffic bursts
✓ Smooth rate control
✓ Best for APIs with variable load

Fixed Window

Simple counter-based approach with fixed time windows. Resets at the start of each window period.

✓ Simple implementation
✓ Low memory usage
✓ Best for strict quotas

Sliding Window

Combines fixed window and sliding log approaches for more accurate rate limiting without edge case issues.

✓ Most accurate
✓ No edge case spikes
✓ Best for consistent limits

Quick Start

Get up and running in 5 minutes

1

Install & Configure

# Clone the repository
git clone https://github.com/PrakarshSingh5/Rate-limiter-service.git
cd rate-limiter-service

# Install dependencies
npm install

# Configure environment
cp .env.example .env
# Edit .env with your Redis configuration
2

Start the Service

# Development mode
npm run dev

# Production mode
npm run build
npm start

# Service runs on:
# REST API: http://localhost:3000
# gRPC: localhost:50051
3

Create a Rate Limit Rule

curl -X POST http://localhost:3000/v1/rules \
  -H "Content-Type: application/json" \
  -d '{
    "name": "API Rate Limit",
    "algorithm": "token_bucket",
    "limit": 100,
    "window": 60,
    "keys": [
      {"type": "user", "value": "*"}
    ]
  }'
4

Check Rate Limit

curl -X POST http://localhost:3000/v1/check \
  -H "Content-Type: application/json" \
  -d '{
    "key": "user:123",
    "endpoint": "/api/data"
  }'

# Response:
{
  "allowed": true,
  "limit": 100,
  "remaining": 99,
  "resetAt": "2024-01-11T03:00:00Z"
}

Integration Examples

Easy integration with any tech stack

Express.js Middleware

const axios = require('axios');

// Rate limiter middleware
async function rateLimiter(req, res, next) {
  try {
    const response = await axios.post('http://localhost:3000/v1/check', {
      key: `user:${req.user.id}`,
      endpoint: req.path
    });

    // Set rate limit headers
    res.set('X-RateLimit-Limit', response.data.limit);
    res.set('X-RateLimit-Remaining', response.data.remaining);
    res.set('X-RateLimit-Reset', response.data.resetAt);

    if (!response.data.allowed) {
      return res.status(429).json({
        error: 'Rate limit exceeded',
        retryAfter: response.data.retryAfter
      });
    }

    next();
  } catch (error) {
    console.error('Rate limiter error:', error);
    next(); // Fail open
  }
}

// Apply to routes
app.use('/api', rateLimiter);

app.get('/api/data', (req, res) => {
  res.json({ message: 'Success!' });
});

Flask Decorator

import requests
from functools import wraps
from flask import request, jsonify

RATE_LIMITER_URL = 'http://localhost:3000/v1/check'

def rate_limit(f):
    @wraps(f)
    def decorated_function(*args, **kwargs):
        user_id = request.headers.get('X-User-ID', 'anonymous')
        
        response = requests.post(RATE_LIMITER_URL, json={
            'key': f'user:{user_id}',
            'endpoint': request.path
        })
        
        data = response.json()
        
        if not data.get('allowed'):
            return jsonify({
                'error': 'Rate limit exceeded',
                'retryAfter': data.get('retryAfter')
            }), 429
        
        return f(*args, **kwargs)
    
    return decorated_function

@app.route('/api/data')
@rate_limit
def get_data():
    return jsonify({'message': 'Success!'})

Go HTTP Middleware

package main

import (
    "bytes"
    "encoding/json"
    "net/http"
)

type CheckRequest struct {
    Key      string `json:"key"`
    Endpoint string `json:"endpoint"`
}

type CheckResponse struct {
    Allowed   bool   `json:"allowed"`
    Limit     int    `json:"limit"`
    Remaining int    `json:"remaining"`
    ResetAt   string `json:"resetAt"`
}

func RateLimiterMiddleware(next http.Handler) http.Handler {
    return http.HandlerFunc(func(w http.ResponseWriter, r *http.Request) {
        userID := r.Header.Get("X-User-ID")
        
        checkReq := CheckRequest{
            Key:      "user:" + userID,
            Endpoint: r.URL.Path,
        }
        
        jsonData, _ := json.Marshal(checkReq)
        resp, err := http.Post(
            "http://localhost:3000/v1/check",
            "application/json",
            bytes.NewBuffer(jsonData),
        )
        
        if err != nil {
            // Fail open
            next.ServeHTTP(w, r)
            return
        }
        defer resp.Body.Close()
        
        var checkResp CheckResponse
        json.NewDecoder(resp.Body).Decode(&checkResp)
        
        if !checkResp.Allowed {
            w.WriteHeader(http.StatusTooManyRequests)
            json.NewEncoder(w).Encode(map[string]string{
                "error": "Rate limit exceeded",
            })
            return
        }
        
        next.ServeHTTP(w, r)
    })
}

Spring Boot Interceptor

import org.springframework.web.servlet.HandlerInterceptor;
import org.springframework.http.HttpStatus;
import org.springframework.web.client.RestTemplate;

public class RateLimiterInterceptor implements HandlerInterceptor {
    
    private final RestTemplate restTemplate = new RestTemplate();
    private static final String RATE_LIMITER_URL = 
        "http://localhost:3000/v1/check";
    
    @Override
    public boolean preHandle(HttpServletRequest request, 
                            HttpServletResponse response, 
                            Object handler) throws Exception {
        
        String userId = request.getHeader("X-User-ID");
        
        CheckRequest checkRequest = new CheckRequest();
        checkRequest.setKey("user:" + userId);
        checkRequest.setEndpoint(request.getRequestURI());
        
        CheckResponse checkResponse = restTemplate.postForObject(
            RATE_LIMITER_URL, 
            checkRequest, 
            CheckResponse.class
        );
        
        if (checkResponse != null && !checkResponse.isAllowed()) {
            response.setStatus(HttpStatus.TOO_MANY_REQUESTS.value());
            response.getWriter().write(
                "{\"error\":\"Rate limit exceeded\"}"
            );
            return false;
        }
        
        return true;
    }
}
